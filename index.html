// AI Chat
const res = await fetch('/api/ai/chatgpt?text=Hello');
const data = await res.json();
console.log(data.result);

// Or with async/await in a function
async function chat(text) {
  const { result } = await (await fetch(`/api/ai/chatgpt?text=${text}`)).json();
  return result;
}
# AI Chat
import requests

  'https://kyoto-rest-api.vercel.app/api/ai/chatgpt',
  params={'text': 'Hello'}
)
data = res.json()
print(data['result'])
